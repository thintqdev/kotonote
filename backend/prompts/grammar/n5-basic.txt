Tạo MỘT bài ngữ pháp JLPT {{jlpt}} (mức phù hợp người mới).

{
  "pattern": "〜…",
  "teaser": { "ja": "", "vi": "" },
  "topicRibbon": { "ja": "", "vi": "" },
  "connection": { "ja": "", "vi": "" },
  "meaning": { "ja": "", "vi": "" },
  "usage": { "ja": "", "vi": "" },
  "usageNote": { "ja": "", "vi": "" },
  "pointBubble": { "ja": "", "vi": "" },
  "examples": [{ "ja": "", "vi": "" }],
  "ng": { "ja": ["..."], "vi": ["..."] },
  "ngNote": { "ja": "", "vi": "" },
  "compare": {
    "caption": { "ja": "", "vi": "" },
    "colLabels": [{ "ja": "", "vi": "" }],
    "rows": [{ "label": { "ja": "", "vi": "" }, "cells": [{ "ja": "", "vi": "" }] }]
  },
  "memo": { "ja": "", "vi": "" },
  "practice": { "items": [{ "ja": "", "vi": "" }] }
}
